/*CRIAÇÃO DE DOMÍNIOS*/
CREATE DOMAIN Login AS VARCHAR(15) NOT NULL; 
CREATE DOMAIN Valor AS FLOAT NOT NULL CHECK (value > 0.0);
CREATE DOMAIN Nome AS VARCHAR(20) NOT NULL;
CREATE DOMAIN Nome_Meta AS VARCHAR(20) NOT NULL;
CREATE DOMAIN Codigo AS INT NOT NULL;

/*CRIAÇÃO DE TABELAS*/
CREATE TABLE USUARIO
(Login VARCHAR,
 Senha VARCHAR(15) NOT NULL,
 Nome_Familia VARCHAR(15) NOT NULL,
 Qtd_Pessoas INT NOT NULL,
 PRIMARY KEY (Login));
 
CREATE TABLE RECEITA
(Nome VARCHAR,
 Codigo INT, 
 Valor FLOAT,
 Data_Rec VARCHAR(10),
 Login VARCHAR,
 PRIMARY KEY(Codigo),
 FOREIGN KEY (Login) REFERENCES USUARIO(Login));

CREATE TABLE EDUCACAO
(Nome VARCHAR,

 Data_Desp VARCHAR(10),
 Login VARCHAR,
